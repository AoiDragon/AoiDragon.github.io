1.POJ-3259 Wormholes

原题链接

判断图中有无负环的模板题。乍一看需要用多元最短路，但实际上spfa就可。因为只要图中存在负环，最短路就一定会不断经过它，因为走一次负环等同于两点间路程的减少。所以可以直接用spfa求解。

当一个点入列超过n-1次时，就说明出现了负环。这是因为不存在负环时，每个点最多在一条最短路中出现一次，而一共只有n-1条单源最短路。大于n-1次说明该点在某条最短路中被经过了多次，一定存在负环。

#include<iostream>
#include<queue>
#include<algorithm>
#include<vector>
#include<utility>
#include<cstring>

using namespace std;

typedef pair<int, int> P;
vector<P> G[2505];
bool vis[505];
int num[505], dis[505];
queue<int> q;
int f, n, m, w;
const int INF=0x3f3f3f3f;

bool spfa()
{
    memset(vis, 0, sizeof(vis));
    memset(num, 0, sizeof(num));
    memset(dis, INF, sizeof(dis));
    q.push(1);
    num[1]++;
    vis[1]=1;
    dis[1]=0;
    while(!q.empty())
    {
        int now=q.front();
        q.pop();
        vis[now]=0;
        for(int i=0; i<G[now].size(); i++)
        {
            int to=G[now][i].first;
            int w=G[now][i].second;
            if(dis[now]+w<dis[to])
            {
                dis[to]=dis[now]+w;
                if(vis[to]==0)
                {
                    q.push(to);
                    vis[to]=1;
                    num[to]++;
                }
            }
            if(num[to]>n-1)
                return true;
        }
    }
    return false;
}

int main(void)
{
    cin>>f;
    while(f--)
    {
        cin>>n>>m>>w;
        for(int i=0; i<=2500; i++)
            G[i].clear();
        while(!q.empty())
            q.pop();
        for(int i=0;i<m; i++)
        {
            int s, e, t;
            cin>>s>>e>>t;
            G[s].push_back({e, t});
            G[e].push_back({s, t});
        }
        for(int i=0; i<w; i++)
        {
            int s, e, t;
            cin>>s>>e>>t;
            G[s].push_back({e, -t});
        }
        if(spfa())
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
    return 0;
}

2.POJ-2139 Six Degrees of Cowvin Bacon

原题链接

多源最短路问题，用Floyd算法，附上一篇讲解较为详细的博客

g[i] [j]代表从第i个点到第j个点且只经过前k个点的最短路。

#include<iostream>
#include<algorithm>
#include<cstring>

using namespace std;

int g[305][305], a[10005], ans=0x3f3f3f3f;

int main(void)
{
    int n, m;
    cin>>n>>m;
    memset(g, 0x3f, sizeof(g));
    for(int i=0; i<=n; i++)
        g[i][i]=0;
    for(int i=0; i<m; i++)
    {
        int k;
        cin>>k;
        for(int j=0; j<k; j++)
            cin>>a[j];
        for(int j=0; j<k-1; j++)
            for(int l=j+1; l<k; l++)
                g[a[j]][a[l]]=g[a[l]][a[j]]=1;
    }
    for(int k=1; k<=n; k++)
        for(int i=1; i<=n; i++)
            for(int j=1; j<=n; j++)
                g[i][j]=min(g[i][j], g[i][k]+g[k][j]);
    for(int i=1; i<=n; i++)
    {
        int temp=0;
        for(int j=1; j<=n; j++)
            temp+=g[i][j];
        ans=min(ans, temp);
    }
    cout<<(100*ans/(n-1))<<endl;
    return 0;
}

3.POJ-1258 Agri-Net

原题链接

最小生成树的模板题。最小生成树有两种常用算法，本题我是用Prim算法写的。

Prim算法的原理是不断枚举和当前的最小生成树这个集合内的点距离(简记为到树的距离)最短的点，将其加入到树中，更新剩余的点到树的距离。不断重复上述过程，直到所有点都被加入树中。

另一种算法是Kruskal算法。首先对边权从小到大排序，依次将其加入树中。在加入前，需要判断加入这条边会不会形成环，这里需要用到并查集。设边e连接顶点u,v，如果u,v之前并未通过一系列边连通在一起，那么这条边的加入就不会形成圈。所以加入一条边后，我们就用并查集将边的顶点放在一类中，再加入新的边时，只需判断两个顶点在已有并查集中是否属于同一类。

#include<iostream>
#include<cstring>
#include<algorithm>

using namespace std;

int g[105][105], vis[105], mincost[105], val, n;

int prim()
{
    int ans=0;
    memset(vis, 0, sizeof(vis));
    memset(mincost, 0x3f, sizeof(mincost));
    mincost[1]=0;//只设置起点的距离，但不要把起点放进树里
    while(true)//直到所有点都被在树中时结束
    {
        int v=-1;
        for(int i=1; i<=n; i++)
            if(!vis[i]&&(v==-1||mincost[v]>mincost[i]))//遍历不在树中的点，并求出与树间距离最小的点
                v=i;
        if(v==-1)//所有的点已在树中
            break;
        ans+=mincost[v];
        vis[v]=1;
        for(int i=1; i<=n; i++)//更新树与其他点间的最小距离
            mincost[i]=min(mincost[i], g[v][i]);
    }
    return ans;
}

int main(void)
{
    while(cin>>n)
    {
        for(int i=1; i<=n; i++)
            for(int j=1; j<=n; j++)
            {
                cin>>val;
                g[i][j]=val;
            }
        cout<<prim()<<endl;
    }
    return 0;
}
