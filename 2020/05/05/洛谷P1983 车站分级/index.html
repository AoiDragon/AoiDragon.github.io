洛谷P1983 车站分级

洛谷 拓扑排序
ACM题解
发布日期:   2020-05-05

原题链接

第一眼看上去我实在没把这道题和拓扑排序想到一起去。但仔细思考一下，题目中要求列车经过的级别大于等于始发站的站点都要停靠，也就是说没有被停靠的站点的级别一定小于停靠的站点。从级别低的站点向级别高的站点连一条单向边，代表二者间的关系，这样我们就得到了一张有向无环图。

现在我们可以用拓扑排序了，但如何将排序和确定级别相关联呢？由于级别高的车站都会有从级别低的车站发出的边，也就是说入度不为0，所以开始时所有入度为0的点便是级别最低的点。将这些点去除并减去与它们相连的点的入度后，剩下所有的点的级别都大于等于2。同理，此时级别最低的点的入度也应为0，这样重复第一遍的过程，就得到了级别为2的点。接下来以此类推，就可以得到所有点的等级。

#include<cstdio>
#include<algorithm>
#include<cstring>
#include<queue>
#include<vector>
#include<utility>

using namespace std;

typedef pair<int, int> p;
int n, m, num, ans;
int vis[1005], s[1005], deg[1005], flag[1005][1005];
vector<int> g[1005];
queue<p> q;

int toposort()
{
    while(!q.empty())
    {
        p now=q.front();
        q.pop();
        int point=now.first, level=now.second;
        for(int i=0; i<g[point].size(); i++)
        {
            deg[g[point][i]]--;
            if(deg[g[point][i]]==0)
                q.push({g[point][i], level+1});
        }
        ans=max(ans, level);
    }
    return ans;
}

int main(void)
{
    scanf("%d%d", &n, &m);
    for(int i=1; i<=m; i++)
    {
        memset(vis, 0, sizeof(vis));
        scanf("%d", &num);//停靠车站数
        for(int j=1; j<=num; j++)
        {
            scanf("%d", &s[j]);
            vis[s[j]]=1;
        }
        for(int j=s[1]; j<=s[num]; j++)
        {
            if(!vis[j])//从未停靠站向停靠站连边
                for(int k=1; k<=num; k++)
                {
                    if(!flag[j][s[k]])//防止重复连边
                    {
                        g[j].push_back(s[k]);
                        deg[s[k]]++;
                        flag[j][s[k]]=1;
                    }
                }
        }
    }
    for(int i=1; i<=n; i++)
        if(deg[i]==0)
            q.push({i, 1});
    printf("%d\n", toposort());
    return 0;
}
