洛谷P1462 通往奥格瑞玛的道路

最短路 洛谷 二分
ACM题解
发布日期:   2020-05-26

原题链接

题目描述有点绕。一条路径中各城市收取的过路费会有一个最大值，求这些最大值的最小值。

一看到最大值的最小值，那肯定需要用到二分，关键就在于怎么把二分和最短路联系起来。

我们可以把二分中判断mid是否符合要求的部分设置为判断当前是否有满足mid值的路径。因为还要求经过的路径权值之和不能超过某个值，那么我们求直接求最短路就可以了。凡是收取的费用大于mid的点，我们在求最短路时直接忽略。假如有一条别的路径上的最大值要比最短路上的最大值要小，那接下来的二分一定会使mid小于最短路上的最大值，从而遍历到更优的结果。

#include<iostream>
#include<algorithm>
#include<queue>
#include<vector>

using namespace std;

typedef pair<int, int> P;
vector<P> g[10005];
queue<int> q;
int vis[10005], dis[10005], n, m, o, f[10005], l, r, mid, ans;

bool spfa(int x)
{
    for(int i=1; i<=n; i++)
    {
        dis[i]=1e9;
        vis[i]=0;
    }
    q.push(1);
    vis[1]=1;
    dis[1]=0;
    while(!q.empty())
    {
        int now=q.front();
        q.pop();
        vis[now]=0;
        for(int i=0; i<g[now].size(); i++)
        {
            int to=g[now][i].first;
            int w=g[now][i].second;
            if(dis[to]>dis[now]+w&&f[to]<=x)
            {
                dis[to]=dis[now]+w;
                if(!vis[to])//这里容易写错
                {
                    q.push(to);
                    vis[to]=1;
                }
            }
        }
    }
    if(dis[n]<o)
        return true;
    return false;
}

int main(void)
{
    cin>>n>>m>>o;
    for(int i=1; i<=n; i++)
    {
        cin>>f[i];
        r=max(r, f[i]);
    }
    for(int i=1; i<=m; i++)
    {
        int a, b, c;
        cin>>a>>b>>c;
        g[a].push_back({b, c});
        g[b].push_back({a, c});
    }
    if(!spfa(r))//如果最大值的最小值取到最大还不行，说明无解
    {
        cout<<"AFK"<<endl;
        return 0;
    }
    while(l<=r)
    {
        mid=(l+r)/2;
        if(spfa(mid))
        {
            ans=mid;//二分的边界条件总感觉容易出错，所以采用了这种保险一点的写法
            r=mid-1;
        }
        else
            l=mid+1;
    }
    cout<<ans<<endl;
    return 0;
}
