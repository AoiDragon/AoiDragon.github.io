洛谷P1278 单词游戏

洛谷 状态压缩dp
ACM题解
发布日期:   2020-05-12

原题链接

给定一系列由字母组成的字符串，将它们拼接在一起。只有当一个字符串末尾的字符与另一个字符串的首字符相同时，二者才可以拼在一起。求拼接后字符串的最大长度。

第一反应是通过dfs递归回溯，但这样会t掉三个点。

有两种解决方法。第一种是假定在时间达到题目要求之前就已经遍历到了最佳答案，这样在dfs函数中添加对于程序运行时间的判断，如果运行达到时间上限，就直接结束程序。这种方法虽然投机取巧，能够AC也是因为题目数据不够好，但有所了解总归是好的。

#include<iostream>
#include<algorithm>
#include<cstring>
#include<ctime>

using namespace std;

string str[20];
char s[20], e[20];
int vis[20], n, ans, temp, tim;

void dfs(int x)
{
    for(int i=0; i<n; i++)
    {
        if(i==x)
            continue;
        if(e[x]==s[i]&&!vis[i])
        {
            vis[i]=1;
            temp+=str[i].size();
            dfs(i);
            temp-=str[i].size();
            vis[i]=0;
        }
    }
    ans=max(ans, temp);
    if(clock()-tim>10000)//超时
    {
        cout<<ans<<endl;
        exit(0);
    }    
    return;
}

int main(void)
{
    tim=clock();//开始计时
    cin>>n;
    for(int i=0; i<n; i++)
    {
        cin>>str[i];
        int a=str[i].size();
        s[i]=str[i][0];//记录每个字符串的首字母
        e[i]=str[i][a-1];//尾字母
    }
    for(int i=0; i<n; i++)
    {
        vis[i]=1;
        temp+=str[i].size();
        dfs(i);
        vis[i]=0;
        temp=0;
    }
    cout<<ans<<endl;
    return 0;
}

正经的解决方法是采用状态压缩dp。一般的dp数组的下标一般是对题目给出的物品各属性的遍历，例如在背包问题中我们第一维遍历物品体积，第二维遍历物品价值。而状态压缩dp将状态作为数组的下标并对其进行遍历。具体实现方式是通过二进制。

如果我们用1表示选取一个物品，0表示未选择一个物品，那么对于在5个物品中选择第1、3、5个的状态就可以表示为10101。我们还可以通过位运算判断一个点是否已经被选取。

if(k& (1 << (i-1) ) )//判断状态k中第i个物品是否被选取

还是以刚才的情况举例，判断第2个物品是否被选取，我们就做10101&10，得到0，说明2号物品未被选取。这样做的原理就是通过右移一定的位数，获得一个除了想验证的位数外全为0的二进制数，这样在与状态做与运算时，其他点结果一定为0。于是如果这一点未被选取过，结果为0&1=0，否则不为0。

本题的dp数组还需要添加一维，表示当前状态最后一次使用的字符串，用于判断能否连接。

#include<iostream>
#include<algorithm>
#include<cstring>

using namespace std;

string str[20];
char s[20], e[20];
int dp[1<<16][20], n, ans;

int main(void)
{
    cin>>n;
    for(int i=1; i<=n; i++)
    {
        cin>>str[i];
        dp[1<<(i-1)][i]=str[i].size();//对只选择一个字符串时的初始化
        int a=str[i].size();
        s[i]=str[i][0];
        e[i]=str[i][a-1];
    }
    for(int k=0; k<(1<<n); k++)
        for(int i=1; i<=n; i++)
            for(int j=1; j<=n; j++)
            {
                if(i==j)
                    continue;
                if(e[i]==s[j]&&!(k&(1<<(j-1))))//能接上且当前组合没有用到j
                    dp[k+(1<<(j-1))][j]=max(dp[k+(1<<(j-1))][j], dp[k][i]+(int)str[j].size());//第一项是已经遍历过的以j结尾的串长，第二项是将j连到新串上的新值
            }
    for(int i=0; i<(1<<n); i++)
        for(int j=1; j<=n; j++)
            ans=max(ans, dp[i][j]);
    cout<<ans<<endl;
    return 0;
}
