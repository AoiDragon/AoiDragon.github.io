1. How Many Answers Are Wrong

原题链接

每一组输入包括三个数据：区间左端点，区间右端点，区间长度。我们需要根据数据判断有几组输入是矛盾的。

有了食物链(POJ1182)的基础，再做这题就比较轻松了。本题的权值就是各节点到根节点的距离。对一棵树上的数据判断是否矛盾，对不在一棵树上的数据合并。

#include<iostream>
#include<cstdio>

using namespace std;

int N, M, a, b, s, ans;
int par[200050];
int w[200050];

int find(int x)
{
    if(par[x]==x)
        return par[x];
    int fx=par[x];
    par[x]=find(par[x]);
    w[x]+=w[fx];
    return par[x];
}

int main(void)
{
    while(scanf("%d%d", &N, &M)!=EOF)
    {
        ans=0;
        for(int i=0; i<=N; i++)
        {
            par[i]=i;
            w[i]=0;
        }
        for(int i=0; i<M; i++)
        {
            cin>>a>>b>>s;
            //这样做是为了让[1,5][6,10]这样的区间可以正常合并
            a--;
            int fa=find(a);
            int fb=find(b);
            if(fa==fb)
            {
                if(w[b]-w[a]!=s)
                    ans++;
            }
            else
            {
                par[fb]=fa;
                w[fb]=w[a]-w[b]+s;
            }
        }   
        cout<<ans<<endl;
    }
    return 0;
}
