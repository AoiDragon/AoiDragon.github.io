HDU 1024 Max Sum Plus Plus

dp 冬训补题
ACM题解
发布日期:   2020-06-19

题目传送门https://vjudge.net/contest/357166#problem/G

题意

给定一个长度为n的整数序列（包含负数），取m个子序列，求这m个子序列和的最大值。

题解

普通的想法是以$dp[i][j]$表示选取了第$j$个数字时，前$j$个数字分成$i$个子序列的情况下的最大值。那我们单独分析这第$j$个数字，有两种情况。第一种是分在了第$i$个分组，那么转移方程为$dp[i][j]=dp[i][j-1]+s[j]$。如果不是分在第$i$个分组，那么一定在前$i-1$个组里。则转移方程为$dp[i][j]=max(dp[i-1][k])+s[j]$,其中$i-1<k<j-1$。因为$i-1$个组至少需要$i-1$个数。

但如果再对k进行遍历，复杂度就会达到$O(n^3)$，明显会超时。所以我们先做第一步优化。在遍历$j$的同时把到当前数为止的$dp$数组最大值求出来。

#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>

using namespace std;

int m, n, dp[500005][1000005], s[1000005], mmax;

int main(void)
{
    while(cin>>m>>n)
    {
        mmax=0;
        memset(dp, 0, sizeof(dp));
        for(int i=1; i<=n; i++)
            scanf("%d", &s[i]);
        for(int i=1; i<=m; i++)
        {
            for(int j=1; j<=n; j++)
            {
                mmax=max(mmax, dp[i-1][j-1]);
                dp[i][j]=max(dp[i][j], max(dp[i][j-1]+s[j], mmax+s[j]));
            }
        }
        cout<<dp[m][n]<<endl;
    }
    return 0;
}

但这样还远远不够，因为题目的数据实在太大了。而$m$的数据并没有给上限，最坏的情况数组会超内存。所以还需要进一步优化。采用类似背包问题的优化方式，将数组的第一维去掉，变成$dp[j]$。第一种情况转移方程变成$dp[j]=dp[j-1]+s[j]$,第二种我们需要用一个数组记录$dp[i-1][k]$的最大值.

#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>

using namespace std;

const int INF=0x3f3f3f3f;
int m, n, dp[10005], s[10005], M[10005], temp;

int main(void)
{
    while(cin>>m>>n)
    {
        memset(dp, 0, sizeof(dp));
        memset(M, 0, sizeof(M));
        for(int i=1; i<=n; i++)
            scanf("%d", &s[i]);
        for(int i=1; i<=m; i++)
        {
            temp=-INF;//可能有负数
            for(int j=i; j<=n; j++)
            {
                dp[j]=max(dp[j-1], M[j-1])+s[j];
                //这里求之前的最大值我绕了好久
                //不清楚为什么只更新到M[j-1],这样下一次循环用到的M[j-1]就
                //相当于这次的M[j],没有更新过
                //但要注意M[j-1]实际上是dp[i-1][k]的最大值，就是需要不更新。
                //已经更新过的M[1...j-2]是dp[i][k]的最大值，在i下个的下个循环中用
                M[j-1]=temp;
                temp=max(temp, dp[j]);
            }
        }
        cout<<dp[n]<<endl;
    }
    return 0;
}
