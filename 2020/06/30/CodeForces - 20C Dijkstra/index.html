CodeForces - 20C Dijkstra

最短路 冬训补题
ACM题解
发布日期:   2020-06-30

题目传送门

题意

给定一个带权无向图。你需要求出从点 1 到点 n的最短路。

第一行有两个正整数 n 和 m (2 ≤ n ≤ $10^5$, 0 ≤ m ≤ $10^5$)，n 表示点数， m 表示边数。接下来的m行中，每行有三个整数：ai, bi 和 wi (1 ≤ ai, bi ≤ n, 1 ≤ wi ≤ $10^6$)，其中 ai, bi为边的两个端点，wi 为边的长度。

图中可能存在环和重边。

如果不存在最短路则输出-1；否则输出对应的最短路径上的点。如果有多个解，输出任意一个就行。

$Example$

$Input $

5 6
1 2 2
2 5 5
2 3 4
1 4 1
4 3 3
3 5 1

$Output $

1 4 3 5

题解

在spfa的模板上做一些修改即可。第一点是添加一个pre数组用来记录每个点前置的点。当点$i$的距离被点$j$更新时，令pre[i]=j。这样跑完一遍最短路后，就可以从终点开始遍历出最短路经过的节点。第二点是数据量比较大，有一些不必要的搜索步骤可以省略。如果dis[i]>dis[n]，那么这个点就不会对最短路的更新做出贡献，可以直接跳过。最后记得dis数组要开long long。

#include<iostream>
#include<algorithm>
#include<vector>
#include<queue>
#include<cstring>

using namespace std;

typedef pair<int, int> P;
int m, n, vis[100005], num[100005], pre[100005], ans[100005];
const long long INF=1e18;
long long dis[100005];
queue<int> q;
vector<P> g[100005];

void dijkstra()
{
    while(!q.empty())
    {
        int now=q.front();
        q.pop();
        vis[now]=0;
        if(dis[now]>dis[n])
            continue;
        for(int i=0; i<g[now].size(); i++)
        {
            int v=g[now][i].first;
            int w=g[now][i].second;
            if(dis[v]>dis[now]+w)
            {
                dis[v]=dis[now]+w;
                pre[v]=now;
                if(!vis[v])
                {
                    q.push(v);
                    vis[v]=1;
                    num[v]++;
                    if(num[v]>=n)
                        return;
                }  
            }
        }
    }
    return;
}

int main(void)
{
    cin>>n>>m;
    for(int i=0; i<=n; i++)
        dis[i]=INF;
    for(int i=0; i<m; i++)
    {
        int a, b, w;
        cin>>a>>b>>w;
        g[a].push_back({b, w});
        g[b].push_back({a, w});
    }
    q.push(1);
    vis[1]=1;
    dis[1]=0;
    num[1]++;
    dijkstra();
    if(dis[n]==INF)
        cout<<-1<<endl;
    else
    {
        int now=n, k=0;
        while(now!=1)
        {
            ans[k]=now;
            now=pre[now];
            k++;
        }
        ans[k]=1;
        for(int i=k; i>=0; i--)
            cout<<ans[i]<<(i==0?'\n':' ');
    }
    return 0;
}
