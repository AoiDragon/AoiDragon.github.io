1.POJ-3176 Cow Bowling

原题链接

#include<iostream>
#include<algorithm>
#include<cstring>

using namespace std;

int s[355][355], dp[355][355];

int main(void)
{
    int n, ans;
    while(cin>>n)
    {
        ans=0;
        memset(s, 0, sizeof(s));
        memset(dp, 0, sizeof(dp));
        for(int i=1; i<=n; i++)//第一行和第一列从下标为1开始，留白，不用对边界特判
            for(int j=1; j<=i; j++)
            {
                cin>>s[i][j];
                dp[i][j]=s[i][j]+max(dp[i-1][j-1], dp[i-1][j]);
            }
        for(int i=1; i<=n; i++)
            ans=max(ans, dp[n][i]);
        cout<<ans<<endl;
    }
    return 0;
}

dp问题的关键就是确定状态方程的值和转移方式。状态方程的值一般与题目要求我们求出的值一致，本题中是途经数字之和的最大值。本题中每一层的状态均有上一层相邻的状态转移而来，选出上一层较大者再加上自身即可。对最后一层的所有结果取最大值就得到最终结果。

2.POJ-2229 Sumsets

原题链接(https://vjudge.net/problem/POJ-2229)

本题实际上是完全背包问题。问题也可以表述为从体积为2^i(i=0,1,2,3……)的物品中任选，是其能够填满一个体积为n的背包，其中每件物品可以选择任意次。

dp[i] [j]代表从前i件物品中选取总体积为j的方案数

for(int i=0; i<num; i++)
    for(int j=1; j<=n; j++)
    {
        dp[i][j]=dp[i-1][j];
        if(j>=w[i])
            dp[i][j]+=dp[i-1][j-w[i]];
    }
//前者是不选第i件物品的情况，此时方案数与在前i-1件物品中选取总体积为j的方案数相同,即dp[i-1][j]
//后者是选择第i件物品的情况，则选取前i-1物品后需要为w[i]预留位置，即dp[i-1][j-w[i]]
//需要对j进行特判，看是否放得下

先写出二维形式有助于理解转移方程的原理，接下来就可以进行状态压缩了。

for(int i=0; i<num; i++)
    for(int j=w[i]; j<=n; j++)
        dp[j]+=dp[j-w[i]];
//如果j
#include<iostream>
#include<algorithm>

using namespace std;

const int MaxN=1e6+5;
int dp[MaxN], w[MaxN], num;

int main(void)
{
    int n;
    cin>>n;
    w[0]=1;
    dp[0]=1;
    for(num=1; w[num-1]<=n; num++)
        w[num]=w[num-1]*2;
    for(int i=0; i<num; i++)
        for(int j=w[i]; j<=n; j++)
            dp[j]=(dp[j]+dp[j-w[i]])%(int)1e9;
    cout<<dp[n]<<endl;
    return 0;
}

3.POJ-2385 Apple Catching

原题链接

题目不难，但这是第一次碰上没做过的dp问题能够自己独立ac的，纪念一下。dp还是要多练才会有题感。

dp[i] [j]代表第i次移动时(准确的说是第i-1次，这里将默认从树1的开始算作第一次移动，方便dp数组对最开始的转状态的转移)，第j个果子掉落后奶牛吃到的果子的最大值。

#include<iostream>
#include<algorithm>

using namespace std;

int pos;
int dp[35][1005], apple[1005];

int main(void)
{
    int t, w;
    cin>>t>>w;
    for(int i=1; i<=t; i++)
        cin>>apple[i];
    pos=0;
    for(int i=1; i<=w+1; i++)
    {
        for(int j=1; j<=t; j++)
        {
            //有两种方式转移到当前状态，不换树和换树
            dp[i][j]=max(dp[i][j-1], dp[i-1][j-1]);           
            if(apple[j]==pos+1)
                dp[i][j]+=1;
        }
        pos=(pos+1)%2;
    }
    cout<<dp[w+1][t]<<endl;
    return 0;
}

4.POJ-3616 Milking Time

原题链接

是之前没见过的dp类型，没鼓捣出来。

对于这种时间安排的问题，无论是贪心还是dp基本上都需要先对给出的时间进行排序。本题将各时间间隔按结束时间先后排序。dp[i]代表前i个时间间隔中获得牛奶的最大值。在转移方程上，先只算自己这个时间间隔内获得的牛奶，再依次遍历这次之前的时间间隔（由于事先的排序，遍历到的都是结束时间早于当前状态的时间间隔）。同时判断能不能兼容。

#include<iostream>
#include<algorithm>

using namespace std;

struct node{
    int a, b, eff;
}farmer[1005];
int dp[1005], res=0;

bool cmp(node x, node y)
{
    return x.b<y.b;
}

int main(void)
{
    int n, m, r;
    cin>>n>>m>>r;
    for(int i=1; i<=m; i++)
        cin>>farmer[i].a>>farmer[i].b>>farmer[i].eff;
    sort(farmer, farmer+m+1, cmp);
    for(int i=1; i<=m; i++)
    {        
        dp[i]=farmer[i].eff;
        for(int j=1; j<i; j++)
            if(farmer[j].b+r<=farmer[i].a)//判断能不能兼容
                dp[i]=max(dp[i], dp[j]+farmer[i].eff);
    }
    for(int i=1; i<=m; i++)
        res=max(res, dp[i]);
    cout<<res<<endl;
    return 0;
}

5.POJ-3280 Cheapest Palindrome

原题链接

对一个字符串进行添加或删除字符操作，使之变为回文字符串。这类问题需要明确，对一个字符的删除或添加实际上是等价的。所以只需要取删除和添加成本中的较小值。

以dp[i] [j]代表从使字符串从第i位到到第j位变为回文字符串的成本最小值。如果当前这一段字符串两侧的字符相同，那么成本与除去两侧字符的字符串相同。如果不同，就选择两侧中处理成本较低的一侧。

#include<iostream>
#include<algorithm>

using namespace std;

char str[2005], ch;
int add, del, w[27], dp[2005][2005];

int main(void)
{
    int n, m;
    cin>>n>>m;
    for(int i=0; i<m; i++)
        cin>>str[i];
    for(int i=0; i<n; i++)
    {
        cin>>ch>>add>>del;
        w[ch-'a']=min(add, del);
    }
    for(int i=m-2; i>=0; i--)
    {
        for(int j=i+1; j<m; j++)
        {
            if(str[i]==str[j])
                dp[i][j]=dp[i+1][j-1];
            else 
                dp[i][j]=min(dp[i+1][j]+w[str[i]-'a'], dp[i][j-1]+w[str[j]-'a']);
        }
    }
    cout<<dp[0][m-1]<<endl;
    return 0;
}
