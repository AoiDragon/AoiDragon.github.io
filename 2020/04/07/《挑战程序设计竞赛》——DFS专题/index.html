《挑战程序设计竞赛》——DFS专题

ACM题解 DFS
《挑战程序设计竞赛》
发布日期:   2020-04-07

1. POJ 1979：Red and Black

原题链接

思路

本题类似书中用做例题的POJ2386，要求我们遍历整张图。由于只需遍历一次，所以我们可以直接将走过的点看作障碍，省去vis数组的使用。

代码

#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>

using namespace std;

int w, h, sx, sy, ans, vis[25][25];
int dx[4]={1, 0, -1, 0};
int dy[4]={0, 1, 0, -1};
char maze[25][25];

void dfs(int x, int y)
{
    maze[x][y]='#';
    for(int i=0; i<4; i++)
    {
        int nx=x+dx[i];
        int ny=y+dy[i];
        if(maze[nx][ny]=='.'&&nx>=0&&ny>=0&&nx<h&&ny<w)
        {
            ans++;
            dfs(nx, ny);
        }
    }
    return;
}

int main(void)
{
    scanf("%d%d", &w, &h);
    while(w!=0||h!=0)
    {
        ans=1;
        memset(maze, 0, sizeof(maze));
        memset(vis, 0, sizeof(vis));
        for(int i=0; i<h; i++)
            scanf("%s", &maze[i]);
        for(int i=0; i<h; i++)
            for(int j=0; j<w; j++)
                if(maze[i][j]=='@')
                {
                    sx=i;
                    sy=j;
                    break;
                }
        dfs(sx, sy);
        printf("%d\n", ans);
        scanf("%d%d", &w, &h);
    }
    return 0;
}

在遍历可能出现的情况之后需要对状态进行回溯。一种方法是本题中使用的，不对改变现有的x,y值，而是新建变量完成后续操作。另一种写法如下：

for (int i = 0; i < 4; i++)
{
    x += dx[i];
    y += dy[i];
    if (maze[x][y] == '.' && x >= 0 && y >= 0 && x < h && y < w)
    {
        ans++;
        dfs(x, y);
    }
    x -= dx[i];
    y -= dy[i];
}

完成对更新后x,y的操作后，一定要让其回复到原来的状态。

2. Aizu 0033 Ball

翻译后的原题链接

代码ver1.0

#include<iostream>
#include<cstring>
#include<algorithm>
#include<deque>

using namespace std;

bool flag;
int num;
deque<int> ball, q1, q2;

void dfs()
{
    if(num==10)
        flag=true;
    int now=ball.front();
    if(q1.empty()||now>q1.back())
    {
        ball.pop_front();
        q1.push_back(now);
        num++;
        dfs();
        num--;
        q1.pop_back();
        ball.push_front(now);
    }
    else if(q2.empty()||now>q2.back())
    {
        ball.pop_front();
        q2.push_back(now);
        num++;
        dfs();
        num--;
        q2.pop_back();
        ball.push_front(now);
    }
}

int main(void)
{
    int n;
    cin>>n;
    while(n--)
    {
        while(!q1.empty())
            q1.pop_back();
        while(!q2.empty())
            q2.pop_back();
        while(!ball.empty())
            ball.pop_back();
        for(int i=0; i<10; i++)
        {
            int temp;
            cin>>temp;
            ball.push_back(temp);
        }
        num=0;
        flag=false;
        dfs();
        if(flag)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
    return 0;
}

自己写的版本。用了双向队列模拟小球的运动，稍微繁琐了些。

比较有收获的一点是明白了如何在判断条件成立后返回。答案是不返回。之前老是栽在这个坑里的原因就是因为递归函数的复杂性，即“你以为你在第二层，实际上你在第五层”。当你千辛万苦找到一个解，赶紧return true后，它并非直接回到了主函数，而是要继续一层一层回溯。所以我们干脆不通过返回值这种方法来告诉程序我们找到了解。直接设置一个全局变量flag，在dfs之前赋值为false，当找到解时改变flag的值。dfs结束后我们对flag的值进行判断即可。

代码ver2.0

#include <iostream>

using namespace std;

int a[11], l = 0, r = 0, flag, n;

int main()
{
    void solve();
    cin >> n;
    while (n--)
    {
        flag = 0;
        for (int i = 0; i < 10; i++)
            cin >> a[i];
        solve();
        if (flag)
            cout << "YES" << endl;
        else
            cout << "NO" << endl;
    }
    return 0;
}
void solve()
{
    int dfs(int i);
    dfs(0);
}
int dfs(int i)
{
    int iv;
    if (i == 10)
    {
        flag = 1;
        return (flag);
    }
    if (a[i] > l)
    {
        iv = l;
        l = a[i];
        dfs(i + 1);
        l = iv;
    }
    if (a[i] > r)
    {
        iv = r;
        r = a[i];
        dfs(i + 1);
        r = iv;
    }
}
