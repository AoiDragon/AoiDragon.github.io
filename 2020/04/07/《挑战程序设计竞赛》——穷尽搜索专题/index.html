1. POJ 1979 Red and Black

原题链接

思路

本题类似书中用做例题的POJ2386，要求我们遍历整张图。由于只需遍历一次，所以我们可以直接将走过的点看作障碍，省去vis数组的使用。

代码

#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>

using namespace std;

int w, h, sx, sy, ans, vis[25][25];
int dx[4]={1, 0, -1, 0};
int dy[4]={0, 1, 0, -1};
char maze[25][25];

void dfs(int x, int y)
{
    maze[x][y]='#';
    for(int i=0; i<4; i++)
    {
        int nx=x+dx[i];
        int ny=y+dy[i];
        if(maze[nx][ny]=='.'&&nx>=0&&ny>=0&&nx<h&&ny<w)
        {
            ans++;
            dfs(nx, ny);
        }
    }
    return;
}

int main(void)
{
    scanf("%d%d", &w, &h);
    while(w!=0||h!=0)
    {
        ans=1;
        memset(maze, 0, sizeof(maze));
        memset(vis, 0, sizeof(vis));
        for(int i=0; i<h; i++)
            scanf("%s", &maze[i]);
        for(int i=0; i<h; i++)
            for(int j=0; j<w; j++)
                if(maze[i][j]=='@')
                {
                    sx=i;
                    sy=j;
                    break;
                }
        dfs(sx, sy);
        printf("%d\n", ans);
        scanf("%d%d", &w, &h);
    }
    return 0;
}

在遍历可能出现的情况之后需要对状态进行回溯。一种方法是本题中使用的，不对改变现有的x,y值，而是新建变量完成后续操作。另一种写法如下：

for (int i = 0; i < 4; i++)
{
    x += dx[i];
    y += dy[i];
    if (maze[x][y] == '.' && x >= 0 && y >= 0 && x < h && y < w)
    {
        ans++;
        dfs(x, y);
    }
    x -= dx[i];
    y -= dy[i];
}

完成对更新后x,y的操作后，一定要让其回复到原来的状态。

2. Aizu 0033 Ball

翻译后的原题链接

代码ver1.0

#include<iostream>
#include<cstring>
#include<algorithm>
#include<deque>

using namespace std;

bool flag;
int num;
deque<int> ball, q1, q2;

void dfs()
{
    if(num==10)
        flag=true;
    int now=ball.front();
    if(q1.empty()||now>q1.back())
    {
        ball.pop_front();
        q1.push_back(now);
        num++;
        dfs();
        num--;
        q1.pop_back();
        ball.push_front(now);
    }
    else if(q2.empty()||now>q2.back())
    {
        ball.pop_front();
        q2.push_back(now);
        num++;
        dfs();
        num--;
        q2.pop_back();
        ball.push_front(now);
    }
}

int main(void)
{
    int n;
    cin>>n;
    while(n--)
    {
        while(!q1.empty())
            q1.pop_back();
        while(!q2.empty())
            q2.pop_back();
        while(!ball.empty())
            ball.pop_back();
        for(int i=0; i<10; i++)
        {
            int temp;
            cin>>temp;
            ball.push_back(temp);
        }
        num=0;
        flag=false;
        dfs();
        if(flag)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
    return 0;
}

自己写的版本。用了双向队列模拟小球的运动，稍微繁琐了些。

比较有收获的一点是明白了如何在判断条件成立后返回。答案是不返回。之前老是栽在这个坑里的原因就是因为递归函数的复杂性，即“你以为你在第二层，实际上你在第五层”。当你千辛万苦找到一个解，赶紧return true后，它并非直接回到了主函数，而是要继续一层一层回溯。所以我们干脆不通过返回值这种方法来告诉程序我们找到了解。直接设置一个全局变量flag，在dfs之前赋值为false，当找到解时改变flag的值。dfs结束后我们对flag的值进行判断即可。

代码ver2.0

#include <iostream>

using namespace std;

int a[11], l = 0, r = 0, flag, n;

int main()
{
    void solve();
    cin >> n;
    while (n--)
    {
        flag = 0;
        for (int i = 0; i < 10; i++)
            cin >> a[i];
        solve();
        if (flag)
            cout << "YES" << endl;
        else
            cout << "NO" << endl;
    }
    return 0;
}
void solve()
{
    int dfs(int i);
    dfs(0);
}
int dfs(int i)
{
    int iv;
    if (i == 10)
    {
        flag = 1;
        return (flag);
    }
    if (a[i] > l)
    {
        iv = l;
        l = a[i];
        dfs(i + 1);
        l = iv;
    }
    if (a[i] > r)
    {
        iv = r;
        r = a[i];
        dfs(i + 1);
        r = iv;
    }
}

网上题解版本 相比自己写的省去许多不必要的模拟过程，值得学习。

3. POJ - 3009 Curling 2.0

原题链接

#include<cstdio>
#include<cstring>
#include<algorithm>

using namespace std;

int w, h, sx, sy, ex, ey, ans, num;
int dx[4]={1, 0, -1, 0};
int dy[4]={0, 1, 0, -1};
int map[25][25];
const int INF=0x3f3f3f3f;

void dfs(int x, int y)
{
    if(num>10)
        return;
    for(int i=0; i<4; i++)
    {
        int flag=0;
        int nx=x;
        int ny=y;
        if(nx+dx[i]>=0&&ny+dy[i]>=0&&map[nx+dx[i]][ny+dy[i]]==1)
            continue;
        for(; ;)
        {
            nx+=dx[i];
            ny+=dy[i];
            if(nx<0||ny<0||nx>=h||ny>=w)
                break;
            if(map[nx][ny]==1)
            {
                flag=1;
                break;
            }
            if(nx==ex&&ny==ey)
            {
                ans=min(ans, num);
                break;
            }
        }
        if(flag)
        {
            map[nx][ny]=0;
            nx-=dx[i];
            ny-=dy[i];
            num++;
            dfs(nx, ny);
            num--;
            nx+=dx[i];
            ny+=dy[i];
            map[nx][ny]=1;
        }
    }   
}

int main(void)
{
    scanf("%d%d", &w, &h);
    while(h!=0||w!=0)
    {
        for(int i=0; i<h; i++)
            for(int j=0; j<w; j++)
            {
                scanf("%d", &map[i][j]);
                if(map[i][j]==2)
                {
                    sx=i;
                    sy=j;
                }
                else if(map[i][j]==3)
                {
                    ex=i;
                    ey=j;
                }
            }
        ans=INF;
        num=1;
        dfs(sx, sy);
        if(ans==INF)
            printf("-1\n");
        else
            printf("%d\n", ans);
        scanf("%d%d", &w, &h);
    }
    return 0;
}

总的来说还是比较正常的，需要递归回溯的DFS，只是每一步都要走到底。唯一出了问题的地方是一开始忘记在判断是否到达终点时数组下标可能出现负数。这种情况在数组下标由变量表示时可能经常出现，需要引起重视。

4. Aizu 0558 Cheese

原题链接

#include<iostream>
#include<algorithm>
#include<queue>
#include<cstring>

using namespace std;

typedef pair<int, int> P;
const int INF=0x3f3f3f3f;
char map[1005][1005];
int vis[1005][1005];
int dx[4]={1, 0, -1, 0};
int dy[4]={0, 1, 0, -1};
int w, h, n, ans, sx, sy, ex, ey;
queue<P> q;

void bfs()
{
    while(!q.empty())
    {
        P now=q.front();
        q.pop();
        for(int i=0; i<4; i++)
        {
            int nx=now.first;
            int ny=now.second;
            nx+=dx[i];
            ny+=dy[i];
            if(nx>=0&&ny>=0&&nx<w&&ny<h&&map[nx][ny]!='X'&&vis[nx][ny]==INF)
            {
                if(nx==ex&&ny==ey)
                {
                    vis[ex][ey]=vis[now.first][now.second]+1;
                    return;
                }
                else
                {
                    q.push({nx, ny});
                    vis[nx][ny]=vis[now.first][now.second]+1;
                }
            }
        }
    }
}

int main(void)
{
    cin>>w>>h>>n;
    for(int i=0; i<w; i++)
        for(int j=0; j<h; j++)
            cin>>map[i][j];
    for(int i=0; i<n; i++)
    {
        for(int j=0; j<w; j++)
            for(int k=0; k<h; k++)
            {
                vis[j][k]=INF;
                if(i==0)
                {
                    if(map[j][k]=='S')
                    {
                        vis[j][k]=0;
                        sx=j;
                        sy=k;
                    }
                }
                if(map[j][k]==i+'0')
                {
                    vis[j][k]=0;
                    sx=j;
                    sy=k;
                }
                if(map[j][k]==i+1+'0')
                {
                    ex=j;
                    ey=k;
                }
            }
        q.push({sx, sy});
        bfs();
        ans+=vis[ex][ey];
        while(!q.empty())
            q.pop();
    }
    cout<<ans<<endl;
    return 0;
}

相当于从S到1、从1到2……从n-1到n做多次BFS再累加结果。要将每一次BFS都看成独立的一次搜索，需要留意对数组的恢复和对点是否能走的判断（起点和之前搜索中经过的点也可以通行）。

5. POJ 3669 Meteor Shower

原题链接

#include<cstdio>
#include<algorithm>
#include<queue>

using namespace std;

typedef pair<int, int> P;
const int INF=0x3f3f3f3f;
int dx[4]={1, 0, -1, 0};
int dy[4]={0, 1, 0, -1};
int map[305][305], vis[305][305];
int n, flag, ans;
queue<P> q;

int bfs()
{
    while(!q.empty())
    {
        P now=q.front();
        q.pop();
        int x=now.first;
        int y=now.second;
        if(map[x][y]==1001)
            return vis[x][y];
        for(int i=0; i<4; i++)
        {
            int nx=x+dx[i];
            int ny=y+dy[i];
            if(nx>=0&&ny>=0&&vis[nx][ny]==INF&&(vis[x][y]<=map[nx][ny]-2||map[nx][ny]==1001))
            {
                q.push({nx, ny});
                vis[nx][ny]=vis[x][y]+1;
            }
        }
    }
    return -1;
}

int main(void)
{
    scanf("%d", &n);
    for(int i=0; i<305; i++)
        for(int j=0; j<305; j++)
        {
            map[i][j]=1001;
            vis[i][j]=INF;
        }
    for(int i=0; i<n; i++)
    {
        int x, y, w;
        scanf("%d%d%d", &x, &y, &w);//cin会超时
        if(x>=1)
            map[x-1][y]=min(w,map[x-1][y]);//防止被后续的更新
        if(y>=1)
            map[x][y-1]=min(w, map[x][y-1]);
        map[x][y]=min(w, map[x][y]);
        map[x+1][y]=min(w, map[x+1][y]);
        map[x][y+1]=min(w, map[x][y+1]);
    }
    q.push({0, 0});
    vis[0][0]=0;
    printf("%d\n", bfs());
    return 0;
}

本质上是BFS。相当于将被波及的区域以坠落时间标记，求从原点走到一个未被标记的点所需的最短时间。首先对地图进行处理。由于坠落区域可能发生重叠，所以要记得取最小值。一开始将每个点初始化为0，但题目中流星是可以在t=0时坠落的（惨 Bessie 惨），所以我换成了恰好大于流星最晚坠落时间的1001，也方便取最小值。还有一点是用cin会超时，必须换成scanf。
